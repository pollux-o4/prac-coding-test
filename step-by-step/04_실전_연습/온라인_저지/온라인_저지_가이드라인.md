### 1. 백준 문제 예시
```python
# 백준 1000: A+B
def baekjoon_1000():
    """
    문제: 두 정수 A와 B를 입력받은 다음, A+B를 출력하는 프로그램을 작성하시오.
    입력: 첫째 줄에 A와 B가 주어진다. (0 < A, B < 10)
    출력: 첫째 줄에 A+B를 출력한다.
    """
    a, b = map(int, input().split())
    print(a + b)

# 백준 2750: 수 정렬하기
def baekjoon_2750():
    """
    문제: N개의 수가 주어졌을 때, 이를 오름차순으로 정렬하는 프로그램을 작성하시오.
    """
    n = int(input())
    numbers = [int(input()) for _ in range(n)]
    numbers.sort()
    for num in numbers:
        print(num)
```

### 2. 프로그래머스 문제 예시
```python
# 프로그래머스: 문자열 압축
def programmers_string_compression(s):
    """
    문제: 문자열 압축
    입력: "aabbaccc"
    출력: 7
    """
    if len(s) <= 1:
        return len(s)
    
    min_length = len(s)
    
    for length in range(1, len(s) // 2 + 1):
        compressed = ""
        current = s[:length]
        count = 1
        
        for i in range(length, len(s), length):
            next_str = s[i:i+length]
            if current == next_str:
                count += 1
            else:
                compressed += (str(count) if count > 1 else "") + current
                current = next_str
                count = 1
        
        compressed += (str(count) if count > 1 else "") + current
        min_length = min(min_length, len(compressed))
    
    return min_length
```

```python
# LeetCode 1: Two Sum
def leetcode_two_sum(nums, target):
    """
    문제: Given an array of integers nums and an integer target, 
    return indices of the two numbers such that they add up to target.
    """
    num_map = {}
    
    for i, num in enumerate(nums):
        complement = target - num
        if complement in num_map:
            return [num_map[complement], i]
        num_map[num] = i
    
    return []

# LeetCode 20: Valid Parentheses
def leetcode_valid_parentheses(s):
    """
    문제: Given a string s containing just the characters '(', ')', '{', '}', '[' and ']', 
    determine if the input string is valid.
    """
    stack = []
    brackets = {')': '(', '}': '{', ']': '['}
    
    for char in s:
        if char in '({[':
            stack.append(char)
        elif char in ')}]':
            if not stack or stack.pop() != brackets[char]:
                return False
    
    return len(stack) == 0
```

### 4. 코드포스 문제 예시
```python
# Codeforces: A. Theatre Square
def codeforces_theatre_square(n, m, a):
    """
    문제: Theatre Square in the capital city of Berland has a rectangular shape 
    with the size n × m meters. On the occasion of the city's anniversary, 
    a decision was taken to pave the Square with square granite flagstones. 
    Each flagstone is of the size a × a.
    """
    import math
    width = math.ceil(n / a)
    height = math.ceil(m / a)
    return width * height
```
