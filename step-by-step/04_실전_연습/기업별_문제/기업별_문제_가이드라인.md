# 문제: 문자열 압축
def kakao_string_compression(s):
    if len(s) <= 1:
        return len(s)
    
    min_length = len(s)
    
    for length in range(1, len(s) // 2 + 1):
        compressed = ""
        current = s[:length]
        count = 1
        
        for i in range(length, len(s), length):
            next_str = s[i:i+length]
            if current == next_str:
                count += 1
            else:
                compressed += (str(count) if count > 1 else "") + current
                current = next_str
                count = 1
        
        compressed += (str(count) if count > 1 else "") + current
        min_length = min(min_length, len(compressed))
    
    return min_length
```

# 문제: 안정적인 배열 정렬
def naver_stable_sort(arr):
    # 안정 정렬 구현
    def merge_sort(arr):
        if len(arr) <= 1:
            return arr
        
        mid = len(arr) // 2
        left = merge_sort(arr[:mid])
        right = merge_sort(arr[mid:])
        
        return merge(left, right)
    
    def merge(left, right):
        result = []
        i = j = 0
        
        while i < len(left) and j < len(right):
            if left[i] <= right[j]:
                result.append(left[i])
                i += 1
            else:
                result.append(right[j])
                j += 1
        
        result.extend(left[i:])
        result.extend(right[j:])
        return result
    
    return merge_sort(arr)
```

# 문제: 시뮬레이션 (로봇 청소기)
def samsung_robot_cleaner(n, m, r, c, d, room):
    # 북, 동, 남, 서
    directions = [(-1, 0), (0, 1), (1, 0), (0, -1)]
    cleaned = 0
    
    while True:
        # 현재 위치 청소
        if room[r][c] == 0:
            room[r][c] = 2
            cleaned += 1
        
        # 주변 확인
        found = False
        for _ in range(4):
            d = (d - 1) % 4
            nr, nc = r + directions[d][0], c + directions[d][1]
            
            if 0 <= nr < n and 0 <= nc < m and room[nr][nc] == 0:
                r, c = nr, nc
                found = True
                break
        
        if not found:
            # 후진
            back_d = (d + 2) % 4
            nr, nc = r + directions[back_d][0], c + directions[back_d][1]
            
            if 0 <= nr < n and 0 <= nc < m and room[nr][nc] != 1:
                r, c = nr, nc
            else:
                break
    
    return cleaned
```

# 문제: 최적화된 해시맵 구현
class GoogleHashMap:
    def __init__(self):
        self.size = 1000
        self.buckets = [[] for _ in range(self.size)]
    
    def _hash(self, key):
        return hash(key) % self.size
    
    def put(self, key, value):
        bucket = self.buckets[self._hash(key)]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket[i] = (key, value)
                return
        bucket.append((key, value))
    
    def get(self, key):
        bucket = self.buckets[self._hash(key)]
        for k, v in bucket:
            if k == key:
                return v
        return None
    
    def remove(self, key):
        bucket = self.buckets[self._hash(key)]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket.pop(i)
                return
```

```python
# 문제: 확장 가능한 캐시 시스템
class MicrosoftCache:
    def __init__(self, capacity):
        self.capacity = capacity
        self.cache = {}
        self.access_order = []
    
    def get(self, key):
        if key in self.cache:
            # 접근 순서 업데이트
            self.access_order.remove(key)
            self.access_order.append(key)
            return self.cache[key]
        return -1
    
    def put(self, key, value):
        if key in self.cache:
            # 기존 값 업데이트
            self.cache[key] = value
            self.access_order.remove(key)
            self.access_order.append(key)
        else:
            # 새 값 추가
            if len(self.cache) >= self.capacity:
                # LRU 제거
                lru_key = self.access_order[0]
                del self.cache[lru_key]
                self.access_order.pop(0)
            
            self.cache[key] = value
            self.access_order.append(key)
```
