## 🎯 문자열 문제의 중요성

문자열 문제는 코딩테스트에서 매우 중요한 유형입니다. 텍스트 처리, 패턴 매칭, 파싱 등 다양한 분야에서 활용되며, 실무에서도 자주 사용되는 핵심 역량입니다.

## 📊 문자열 문제의 특징

### 1. 장점
- **실용성**: 실제 개발에서 자주 사용
- **다양성**: 다양한 알고리즘 적용 가능
- **직관성**: 문제 이해가 상대적으로 쉬움

### 2. 단점
- **세부사항**: 인덱스 관리가 복잡
- **경계값**: 문자열 길이와 인덱스 주의
- **메모리**: 대용량 문자열 처리 시 메모리 고려

### 3. 핵심 능력
- **문자열 조작**: 슬라이싱, 연결, 변환
- **패턴 매칭**: 정규표현식, 문자열 검색
- **파싱**: 문자열을 구조화된 데이터로 변환

```python
def is_palindrome(s):
    # 대소문자 구분 없이, 공백과 특수문자 제거
    cleaned = ''.join(c.lower() for c in s if c.isalnum())
    return cleaned == cleaned[::-1]

# 문자열 압축
def compress_string(s):
    if not s:
        return s
    
    result = []
    count = 1
    current = s[0]
    
    for char in s[1:]:
        if char == current:
            count += 1
        else:
            result.append(current)
            if count > 1:
                result.append(str(count))
            current = char
            count = 1
    
    result.append(current)
    if count > 1:
        result.append(str(count))
    
    return ''.join(result)

# 문자열에서 단어 개수 세기
def count_words(s):
    if not s.strip():
        return 0
    return len(s.split())

# 문자열에서 가장 많이 등장하는 문자 찾기
def most_frequent_char(s):
    if not s:
        return None
    
    char_count = {}
    for char in s:
        char_count[char] = char_count.get(char, 0) + 1
    
    return max(char_count.items(), key=lambda x: x[1])[0]
```

```python
# 단순 문자열 검색 (브루트 포스)
def simple_search(text, pattern):
    n, m = len(text), len(pattern)
    
    for i in range(n - m + 1):
        if text[i:i+m] == pattern:
            return i
    
    return -1

# KMP 알고리즘
def kmp_search(text, pattern):
    def compute_lps(pattern):
        lps = [0] * len(pattern)
        length = 0
        i = 1
        
        while i < len(pattern):
            if pattern[i] == pattern[length]:
                length += 1
                lps[i] = length
                i += 1
            else:
                if length != 0:
                    length = lps[length - 1]
                else:
                    lps[i] = 0
                    i += 1
        
        return lps
    
    if not pattern:
        return 0
    
    lps = compute_lps(pattern)
    i = j = 0
    
    while i < len(text):
        if pattern[j] == text[i]:
            i += 1
            j += 1
        
        if j == len(pattern):
            return i - j
        elif i < len(text) and pattern[j] != text[i]:
            if j != 0:
                j = lps[j - 1]
            else:
                i += 1
    
    return -1

# 라빈-카프 알고리즘
def rabin_karp_search(text, pattern):
    def hash_function(s, base=256, mod=101):
        hash_val = 0
        for char in s:
            hash_val = (hash_val * base + ord(char)) % mod
        return hash_val
    
    n, m = len(text), len(pattern)
    if m > n:
        return -1
    
    pattern_hash = hash_function(pattern)
    text_hash = hash_function(text[:m])
    
    # 첫 번째 윈도우 검사
    if pattern_hash == text_hash and text[:m] == pattern:
        return 0
    
    # 슬라이딩 윈도우
    base = 256
    mod = 101
    power = pow(base, m - 1, mod)
    
    for i in range(1, n - m + 1):
        # 이전 해시에서 첫 번째 문자 제거
        text_hash = (text_hash - ord(text[i - 1]) * power) % mod
        # 새 문자 추가
        text_hash = (text_hash * base + ord(text[i + m - 1])) % mod
        
        if pattern_hash == text_hash and text[i:i+m] == pattern:
            return i
    
    return -1
```

```python
# CSV 파싱
def parse_csv(csv_string):
    lines = csv_string.strip().split('\n')
    result = []
    
    for line in lines:
        # 쉼표로 분할 (따옴표 처리 포함)
        fields = []
        current_field = ""
        in_quotes = False
        
        for char in line:
            if char == '"':
                in_quotes = not in_quotes
            elif char == ',' and not in_quotes:
                fields.append(current_field)
                current_field = ""
            else:
                current_field += char
        
        fields.append(current_field)
        result.append(fields)
    
    return result

# JSON 파싱 (간단한 버전)
def parse_json(json_string):
    def parse_value(s, start):
        s = s.strip()
        
        if s[start] == '{':
            return parse_object(s, start)
        elif s[start] == '[':
            return parse_array(s, start)
        elif s[start] == '"':
            return parse_string(s, start)
        elif s[start].isdigit() or s[start] == '-':
            return parse_number(s, start)
        elif s[start:start+4] == 'true':
            return True, start + 4
        elif s[start:start+5] == 'false':
            return False, start + 5
        elif s[start:start+4] == 'null':
            return None, start + 4
    
    def parse_string(s, start):
        end = start + 1
        while end < len(s) and s[end] != '"':
            if s[end] == '\\':
                end += 2
            else:
                end += 1
        return s[start+1:end], end + 1
    
    def parse_number(s, start):
        end = start
        while end < len(s) and (s[end].isdigit() or s[end] in '.-'):
            end += 1
        return float(s[start:end]), end
    
    def parse_object(s, start):
        result = {}
        pos = start + 1
        
        while pos < len(s) and s[pos] != '}':
            if s[pos] == '"':
                key, pos = parse_string(s, pos)
                pos = s.find(':', pos) + 1
                value, pos = parse_value(s, pos)
                result[key] = value
            else:
                pos += 1
        
        return result, pos + 1
    
    def parse_array(s, start):
        result = []
        pos = start + 1
        
        while pos < len(s) and s[pos] != ']':
            if s[pos] != ',':
                value, pos = parse_value(s, pos)
                result.append(value)
            else:
                pos += 1
        
        return result, pos + 1
    
    return parse_value(json_string, 0)[0]
```

```python
# 시저 암호
def caesar_cipher(text, shift, decrypt=False):
    if decrypt:
        shift = -shift
    
    result = ""
    for char in text:
        if char.isalpha():
            ascii_offset = ord('A') if char.isupper() else ord('a')
            shifted = (ord(char) - ascii_offset + shift) % 26
            result += chr(shifted + ascii_offset)
        else:
            result += char
    
    return result

# 비트 암호화
def xor_cipher(text, key):
    result = ""
    for i, char in enumerate(text):
        key_char = key[i % len(key)]
        result += chr(ord(char) ^ ord(key_char))
    return result
```

```python
# Run-Length Encoding (RLE)
def run_length_encode(s):
    if not s:
        return s
    
    result = []
    count = 1
    current = s[0]
    
    for char in s[1:]:
        if char == current:
            count += 1
        else:
            result.append(current + str(count))
            current = char
            count = 1
    
    result.append(current + str(count))
    return ''.join(result)

def run_length_decode(s):
    if not s:
        return s
    
    result = ""
    i = 0
    
    while i < len(s):
        char = s[i]
        i += 1
        
        # 숫자 부분 찾기
        count = ""
        while i < len(s) and s[i].isdigit():
            count += s[i]
            i += 1
        
        if count:
            result += char * int(count)
        else:
            result += char
    
    return result

# Huffman 코딩
import heapq
from collections import Counter

class HuffmanNode:
    def __init__(self, char, freq):
        self.char = char
        self.freq = freq
        self.left = None
        self.right = None
    
    def __lt__(self, other):
        return self.freq < other.freq

def build_huffman_tree(text):
    # 빈도수 계산
    freq = Counter(text)
    
    # 최소 힙 생성
    heap = [HuffmanNode(char, freq) for char, freq in freq.items()]
    heapq.heapify(heap)
    
    # 트리 구성
    while len(heap) > 1:
        left = heapq.heappop(heap)
        right = heapq.heappop(heap)
        
        internal = HuffmanNode(None, left.freq + right.freq)
        internal.left = left
        internal.right = right
        
        heapq.heappush(heap, internal)
    
    return heap[0]

def get_huffman_codes(root, code="", codes=None):
    if codes is None:
        codes = {}
    
    if root.char is not None:
        codes[root.char] = code
    else:
        get_huffman_codes(root.left, code + "0", codes)
        get_huffman_codes(root.right, code + "1", codes)
    
    return codes

def huffman_encode(text):
    if not text:
        return "", None
    
    root = build_huffman_tree(text)
    codes = get_huffman_codes(root)
    
    encoded = ""
    for char in text:
        encoded += codes[char]
    
    return encoded, codes

def huffman_decode(encoded_text, codes):
    if not encoded_text:
        return ""
    
    # 역방향 코드 맵 생성
    reverse_codes = {code: char for char, code in codes.items()}
    
    decoded = ""
    current_code = ""
    
    for bit in encoded_text:
        current_code += bit
        if current_code in reverse_codes:
            decoded += reverse_codes[current_code]
            current_code = ""
    
    return decoded
```
