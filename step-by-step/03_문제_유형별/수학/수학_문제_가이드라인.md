## 🎯 수학 문제의 중요성

수학 문제는 코딩테스트에서 매우 중요한 유형입니다. 수학적 사고와 알고리즘을 결합하여 효율적인 해결책을 찾는 능력을 평가하며, 많은 고급 알고리즘의 기반이 됩니다.

### 1. 장점
- **논리적 사고**: 수학적 증명과 논리
- **효율성**: 수학적 공식으로 빠른 해결
- **창의성**: 다양한 접근 방법

### 2. 단점

- **증명의 어려움**: 수학적 증명이 복잡

### 1. 소수 판별과 소수 생성
```python
import math

# 소수 판별 (O(√n))
def is_prime(n):
    if n < 2:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    
    for i in range(3, int(math.sqrt(n)) + 1, 2):
        if n % i == 0:
            return False
    return True

# 에라토스테네스의 체 (O(n log log n))
def sieve_of_eratosthenes(n):
    is_prime = [True] * (n + 1)
    is_prime[0] = is_prime[1] = False
    
    for i in range(2, int(math.sqrt(n)) + 1):
        if is_prime[i]:
            for j in range(i * i, n + 1, i):
                is_prime[j] = False
    
    return [i for i in range(n + 1) if is_prime[i]]

# 소인수분해
def prime_factorization(n):
    factors = []
    d = 2
    
    while d * d <= n:
        while n % d == 0:
            factors.append(d)
            n //= d
        d += 1
    
    if n > 1:
        factors.append(n)
    
    return factors

# 소인수분해 (개수 포함)
def prime_factorization_with_count(n):
    factors = {}
    d = 2
    
    while d * d <= n:
        while n % d == 0:
            factors[d] = factors.get(d, 0) + 1
            n //= d
        d += 1
    
    if n > 1:
        factors[n] = factors.get(n, 0) + 1
    
    return factors
```

### 2. 최대공약수와 최소공배수
```python
import math

# 유클리드 호제법 (최대공약수)
def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

# 재귀적 구현
def gcd_recursive(a, b):
    if b == 0:
        return a
    return gcd_recursive(b, a % b)

# 최소공배수
def lcm(a, b):
    return abs(a * b) // gcd(a, b)

# 여러 수의 최대공약수
def gcd_multiple(numbers):
    result = numbers[0]
    for num in numbers[1:]:
        result = gcd(result, num)
    return result

# 여러 수의 최소공배수
def lcm_multiple(numbers):
    result = numbers[0]
    for num in numbers[1:]:
        result = lcm(result, num)
    return result
```

### 3. 조합과 순열
```python
import math

# 팩토리얼
def factorial(n):
    if n <= 1:
        return 1
    return n * factorial(n - 1)

# 조합 (nCr)
def combination(n, r):
    if r > n:
        return 0
    if r == 0 or r == n:
        return 1
    
    # 파스칼의 삼각형 이용
    dp = [[0] * (r + 1) for _ in range(n + 1)]
    
    for i in range(n + 1):
        for j in range(min(i, r) + 1):
            if j == 0 or j == i:
                dp[i][j] = 1
            else:
                dp[i][j] = dp[i-1][j-1] + dp[i-1][j]
    
    return dp[n][r]

# 순열 (nPr)
def permutation(n, r):
    if r > n:
        return 0
    return factorial(n) // factorial(n - r)

# 중복 조합 (nHr)
def combination_with_repetition(n, r):
    return combination(n + r - 1, r)
```

### 1. 모듈로 연산
```python
# 모듈로 거듭제곱 (빠른 거듭제곱)
def mod_pow(base, exponent, modulus):
    if modulus == 1:
        return 0
    
    result = 1
    base = base % modulus
    
    while exponent > 0:
        if exponent % 2 == 1:
            result = (result * base) % modulus
        exponent = exponent >> 1
        base = (base * base) % modulus
    
    return result

# 모듈로 역원 (확장 유클리드 호제법)
def mod_inverse(a, m):
    def extended_gcd(a, b):
        if a == 0:
            return b, 0, 1
        else:
            gcd, x, y = extended_gcd(b % a, a)
            return gcd, y - (b // a) * x, x
    
    gcd, x, y = extended_gcd(a, m)
    if gcd != 1:
        raise ValueError("Modular inverse does not exist")
    else:
        return (x % m + m) % m

# 중국인의 나머지 정리
def chinese_remainder_theorem(remainders, moduli):
    def extended_gcd(a, b):
        if a == 0:
            return b, 0, 1
        else:
            gcd, x, y = extended_gcd(b % a, a)
            return gcd, y - (b // a) * x, x
    
    def mod_inverse(a, m):
        gcd, x, y = extended_gcd(a, m)
        if gcd != 1:
            raise ValueError("Modular inverse does not exist")
        else:
            return (x % m + m) % m
    
    total = 0
    product = 1
    
    for modulus in moduli:
        product *= modulus
    
    for remainder, modulus in zip(remainders, moduli):
        pi = product // modulus
        total += remainder * pi * mod_inverse(pi, modulus)
    
    return total % product
```

### 2. 확률과 통계
```python
import math

# 이항계수 확률
def binomial_probability(n, k, p):
    return combination(n, k) * (p ** k) * ((1 - p) ** (n - k))

# 기댓값 계산
def expected_value(values, probabilities):
    return sum(v * p for v, p in zip(values, probabilities))

# 분산 계산
def variance(values, probabilities):
    mean = expected_value(values, probabilities)
    return sum(p * (v - mean) ** 2 for v, p in zip(values, probabilities))

# 표준편차 계산
def standard_deviation(values, probabilities):
    return math.sqrt(variance(values, probabilities))
```

### 3. 기하학 문제

```python
import math

# 두 점 사이의 거리
def distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

# 삼각형의 넓이 (헤론의 공식)
def triangle_area(a, b, c):
    s = (a + b + c) / 2
    return math.sqrt(s * (s - a) * (s - b) * (s - c))

# 원의 넓이
def circle_area(radius):
    return math.pi * radius ** 2

# 원의 둘레
def circle_circumference(radius):
    return 2 * math.pi * radius

# 점이 다각형 내부에 있는지 확인 (레이 캐스팅)
def point_in_polygon(point, polygon):
    x, y = point
    n = len(polygon)
    inside = False
    
    p1x, p1y = polygon[0]
    for i in range(n + 1):
        p2x, p2y = polygon[i % n]
        if y > min(p1y, p2y):
            if y <= max(p1y, p2y):
                if x <= max(p1x, p2x):
                    if p1y != p2y:
                        xinters = (y - p1y) * (p2x - p1x) / (p2y - p1y) + p1x
                    if p1x == p2x or x <= xinters:
                        inside = not inside
        p1x, p1y = p2x, p2y
    
    return inside
```

```python
# 행렬 곱셈
def matrix_multiply(a, b):
    rows_a, cols_a = len(a), len(a[0])
    rows_b, cols_b = len(b), len(b[0])
    
    if cols_a != rows_b:
        raise ValueError("Matrix dimensions do not match")
    
    result = [[0] * cols_b for _ in range(rows_a)]
    
    for i in range(rows_a):
        for j in range(cols_b):
            for k in range(cols_a):
                result[i][j] += a[i][k] * b[k][j]
    
    return result

# 행렬 거듭제곱
def matrix_power(matrix, power):
    if power == 1:
        return matrix
    
    if power % 2 == 0:
        half = matrix_power(matrix, power // 2)
        return matrix_multiply(half, half)
    else:
        return matrix_multiply(matrix, matrix_power(matrix, power - 1))

# 행렬식 계산 (2x2)
def determinant_2x2(matrix):
    return matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[1][0]

# 행렬식 계산 (3x3)
def determinant_3x3(matrix):
    return (matrix[0][0] * (matrix[1][1] * matrix[2][2] - matrix[1][2] * matrix[2][1]) -
            matrix[0][1] * (matrix[1][0] * matrix[2][2] - matrix[1][2] * matrix[2][0]) +
            matrix[0][2] * (matrix[1][0] * matrix[2][1] - matrix[1][1] * matrix[2][0]))
```
