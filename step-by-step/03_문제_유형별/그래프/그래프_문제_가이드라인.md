# 그래프 문제 학습 가이드라인

## 🎯 그래프 문제의 중요성

그래프 문제는 코딩테스트에서 가장 중요한 유형 중 하나입니다. 네트워크, 지도, 소셜 네트워크 등 다양한 현실 세계 문제를 모델링하는 데 사용되며, 많은 고급 알고리즘의 기반이 됩니다.

## 📊 그래프 문제의 특징

### 1. 장점
- **실용성**: 현실 세계 문제 모델링
- **다양성**: 다양한 알고리즘 적용 가능
- **확장성**: 복잡한 문제도 단순한 구조로 표현

### 2. 단점
- **복잡성**: 그래프 구조 이해 필요
- **메모리**: 대용량 그래프 처리 시 메모리 고려
- **알고리즘**: 다양한 그래프 알고리즘 숙지 필요

### 3. 핵심 능력
- **그래프 모델링**: 문제를 그래프로 변환
- **탐색 알고리즘**: DFS, BFS 활용
- **최적화**: 최단 경로, 최소 신장 트리

```python
from collections import defaultdict

# 인접 리스트 (가장 일반적)
class Graph:
    def __init__(self):
        self.graph = defaultdict(list)
    
    def add_edge(self, u, v, weight=1):
        self.graph[u].append((v, weight))
    
    def get_neighbors(self, vertex):
        return self.graph[vertex]
    
    def get_vertices(self):
        return list(self.graph.keys())

# 인접 행렬
class AdjacencyMatrix:
    def __init__(self, vertices):
        self.vertices = vertices
        self.matrix = [[0] * vertices for _ in range(vertices)]
    
    def add_edge(self, u, v, weight=1):
        self.matrix[u][v] = weight
        # 무방향 그래프인 경우
        # self.matrix[v][u] = weight
    
    def has_edge(self, u, v):
        return self.matrix[u][v] != 0
    
    def get_weight(self, u, v):
        return self.matrix[u][v]

# 간선 리스트
class EdgeList:
    def __init__(self):
        self.edges = []
    
    def add_edge(self, u, v, weight=1):
        self.edges.append((u, v, weight))
    
    def get_edges(self):
        return self.edges
```

```python
from collections import deque

# 깊이 우선 탐색 (DFS)
def dfs(graph, start, visited=None):
    if visited is None:
        visited = set()
    
    visited.add(start)
    print(start, end=' ')
    
    for neighbor, weight in graph.get_neighbors(start):
        if neighbor not in visited:
            dfs(graph, neighbor, visited)

# 너비 우선 탐색 (BFS)
def bfs(graph, start):
    visited = set()
    queue = deque([start])
    visited.add(start)
    
    while queue:
        vertex = queue.popleft()
        print(vertex, end=' ')
        
        for neighbor, weight in graph.get_neighbors(vertex):
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

# 연결 요소 개수 세기
def count_connected_components(graph):
    visited = set()
    count = 0
    
    for vertex in graph.get_vertices():
        if vertex not in visited:
            dfs(graph, vertex, visited)
            count += 1
    
    return count

# 사이클 감지 (무방향 그래프)
def has_cycle_undirected(graph):
    visited = set()
    
    def dfs_with_parent(vertex, parent):
        visited.add(vertex)
        
        for neighbor, weight in graph.get_neighbors(vertex):
            if neighbor not in visited:
                if dfs_with_parent(neighbor, vertex):
                    return True
            elif neighbor != parent:
                return True
        
        return False
    
    for vertex in graph.get_vertices():
        if vertex not in visited:
            if dfs_with_parent(vertex, None):
                return True
    
    return False
```

### 3. 최단 경로 알고리즘
```python
import heapq

# 다익스트라 알고리즘 (단일 출발점 최단 경로)
def dijkstra(graph, start):
    distances = {vertex: float('inf') for vertex in graph.get_vertices()}
    distances[start] = 0
    
    pq = [(0, start)]
    
    while pq:
        current_distance, current_vertex = heapq.heappop(pq)
        
        if current_distance > distances[current_vertex]:
            continue
        
        for neighbor, weight in graph.get_neighbors(current_vertex):
            distance = current_distance + weight
            
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(pq, (distance, neighbor))
    
    return distances

# 플로이드-워셜 알고리즘 (모든 쌍 최단 경로)
def floyd_warshall(graph):
    vertices = graph.get_vertices()
    n = len(vertices)
    
    # 거리 행렬 초기화
    distances = [[float('inf')] * n for _ in range(n)]
    
    # 자기 자신으로의 거리는 0
    for i in range(n):
        distances[i][i] = 0
    
    # 직접 연결된 간선들
    for u in vertices:
        for v, weight in graph.get_neighbors(u):
            distances[u][v] = weight
    
    # 플로이드-워셜 알고리즘
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if distances[i][k] + distances[k][j] < distances[i][j]:
                    distances[i][j] = distances[i][k] + distances[k][j]
    
    return distances

# 벨만-포드 알고리즘 (음수 가중치 허용)
def bellman_ford(graph, start):
    vertices = graph.get_vertices()
    distances = {vertex: float('inf') for vertex in vertices}
    distances[start] = 0
    
    # V-1번 반복
    for _ in range(len(vertices) - 1):
        for u in vertices:
            for v, weight in graph.get_neighbors(u):
                if distances[u] + weight < distances[v]:
                    distances[v] = distances[u] + weight
    
    # 음수 사이클 검사
    for u in vertices:
        for v, weight in graph.get_neighbors(u):
            if distances[u] + weight < distances[v]:
                return None  # 음수 사이클 존재
    
    return distances
```

### 1. 최소 신장 트리 (MST)
```python
# 크루스칼 알고리즘
class DisjointSet:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
    
    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]
    
    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py:
            return False
        
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        
        self.parent[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1
        return True

def kruskal_mst(graph):
    edges = []
    for u in graph.get_vertices():
        for v, weight in graph.get_neighbors(u):
            edges.append((weight, u, v))
    
    edges.sort()  # 가중치로 정렬
    
    ds = DisjointSet(len(graph.get_vertices()))
    mst = []
    total_weight = 0
    
    for weight, u, v in edges:
        if ds.union(u, v):
            mst.append((u, v, weight))
            total_weight += weight
    
    return mst, total_weight

# 프림 알고리즘
def prim_mst(graph, start):
    vertices = graph.get_vertices()
    visited = set()
    mst = []
    total_weight = 0
    
    pq = [(0, start, None)]  # (weight, vertex, parent)
    
    while pq and len(visited) < len(vertices):
        weight, vertex, parent = heapq.heappop(pq)
        
        if vertex in visited:
            continue
        
        visited.add(vertex)
        if parent is not None:
            mst.append((parent, vertex, weight))
            total_weight += weight
        
        for neighbor, edge_weight in graph.get_neighbors(vertex):
            if neighbor not in visited:
                heapq.heappush(pq, (edge_weight, neighbor, vertex))
    
    return mst, total_weight
```

```python
# 위상 정렬 (DFS 기반)
def topological_sort_dfs(graph):
    visited = set()
    temp_visited = set()
    result = []
    
    def dfs(vertex):
        if vertex in temp_visited:
            raise ValueError("Cycle detected")
        if vertex in visited:
            return
        
        temp_visited.add(vertex)
        
        for neighbor, weight in graph.get_neighbors(vertex):
            dfs(neighbor)
        
        temp_visited.remove(vertex)
        visited.add(vertex)
        result.append(vertex)
    
    for vertex in graph.get_vertices():
        if vertex not in visited:
            dfs(vertex)
    
    return result[::-1]  # 역순으로 반환

# 위상 정렬 (Kahn's algorithm)
def topological_sort_kahn(graph):
    # 진입 차수 계산
    in_degree = {vertex: 0 for vertex in graph.get_vertices()}
    
    for vertex in graph.get_vertices():
        for neighbor, weight in graph.get_neighbors(vertex):
            in_degree[neighbor] += 1
    
    # 진입 차수가 0인 정점들을 큐에 추가
    queue = deque([vertex for vertex, degree in in_degree.items() if degree == 0])
    result = []
    
    while queue:
        vertex = queue.popleft()
        result.append(vertex)
        
        for neighbor, weight in graph.get_neighbors(vertex):
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)
    
    if len(result) != len(graph.get_vertices()):
        raise ValueError("Cycle detected")
    
    return result
```

### 3. 강한 연결 요소 (SCC)
```python
# 타잔 알고리즘
def tarjan_scc(graph):
    visited = set()
    on_stack = set()
    low = {}
    disc = {}
    time = 0
    sccs = []
    stack = []
    
    def dfs(vertex):
        nonlocal time
        visited.add(vertex)
        on_stack.add(vertex)
        stack.append(vertex)
        disc[vertex] = low[vertex] = time
        time += 1
        
        for neighbor, weight in graph.get_neighbors(vertex):
            if neighbor not in visited:
                dfs(neighbor)
                low[vertex] = min(low[vertex], low[neighbor])
            elif neighbor in on_stack:
                low[vertex] = min(low[vertex], disc[neighbor])
        
        if low[vertex] == disc[vertex]:
            scc = []
            while True:
                w = stack.pop()
                on_stack.remove(w)
                scc.append(w)
                if w == vertex:
                    break
            sccs.append(scc)
    
    for vertex in graph.get_vertices():
        if vertex not in visited:
            dfs(vertex)
    
    return sccs
```

### 4. 이분 매칭
```python
# 이분 매칭 (DFS 기반)
def bipartite_matching(graph, left_vertices, right_vertices):
    match = {v: None for v in right_vertices}
    visited = set()
    
    def dfs(vertex):
        for neighbor in graph.get_neighbors(vertex):
            if neighbor in visited:
                continue
            
            visited.add(neighbor)
            
            if match[neighbor] is None or dfs(match[neighbor]):
                match[neighbor] = vertex
                return True
        
        return False
    
    max_matching = 0
    for vertex in left_vertices:
        visited.clear()
        if dfs(vertex):
            max_matching += 1
    
    return max_matching, match
```

## 📊 그래프 문제의 성능

### 시간복잡도
- **DFS/BFS**: O(V + E)
- **다익스트라**: O((V + E) log V)
- **플로이드-워셜**: O(V³)
- **크루스칼**: O(E log E)
- **프림**: O(E log V)

### 공간복잡도
- **인접 리스트**: O(V + E)
- **인접 행렬**: O(V²)
- **간선 리스트**: O(E)
