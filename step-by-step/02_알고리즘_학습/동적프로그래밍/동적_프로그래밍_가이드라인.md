# 동적 프로그래밍 학습 가이드라인

## 🎯 동적 프로그래밍의 중요성

동적 프로그래밍(DP)은 코딩테스트에서 가장 중요한 알고리즘 중 하나입니다. 복잡한 문제를 작은 하위 문제로 나누어 해결하는 방식으로, 최적화 문제와 조합 문제에서 매우 효과적입니다.

## 📊 동적 프로그래밍의 핵심 개념

### 1. 중복되는 부분 문제 (Overlapping Subproblems)
- 같은 계산이 여러 번 반복되는 경우
- 메모이제이션으로 중복 계산 방지

### 2. 최적 부분 구조 (Optimal Substructure)
- 전체 문제의 최적해가 부분 문제의 최적해로 구성
- 그리디와의 차이점: 모든 부분 문제를 고려

### 3. 메모이제이션 vs 타뷸레이션
- **메모이제이션**: Top-down 접근 (재귀)
- **타뷸레이션**: Bottom-up 접근 (반복)

## 🔧 기본 DP 문제들

### 1. 피보나치 수열
```python
# 기본 재귀 (비효율적)
def fib_recursive(n):
    if n <= 1:
        return n
    return fib_recursive(n-1) + fib_recursive(n-2)

# 메모이제이션 (Top-down)
def fib_memoization(n, memo=None):
    if memo is None:
        memo = {}
    
    if n in memo:
        return memo[n]
    
    if n <= 1:
        return n
    
    memo[n] = fib_memoization(n-1, memo) + fib_memoization(n-2, memo)
    return memo[n]

# 타뷸레이션 (Bottom-up)
def fib_tabulation(n):
    if n <= 1:
        return n
    
    dp = [0] * (n + 1)
    dp[1] = 1
    
    for i in range(2, n + 1):
        dp[i] = dp[i-1] + dp[i-2]
    
    return dp[n]

# 공간 최적화
def fib_optimized(n):
    if n <= 1:
        return n
    
    prev, curr = 0, 1
    for i in range(2, n + 1):
        prev, curr = curr, prev + curr
    
    return curr
```

### 3. 최대 부분 배열 합 (Kadane's Algorithm)
```python
def max_subarray_sum(arr):
    if not arr:
        return 0
    
    max_sum = curr_sum = arr[0]
    
    for num in arr[1:]:
        curr_sum = max(num, curr_sum + num)
        max_sum = max(max_sum, curr_sum)
    
    return max_sum

# DP를 이용한 구현
def max_subarray_sum_dp(arr):
    if not arr:
        return 0
    
    n = len(arr)
    dp = [0] * n
    dp[0] = arr[0]
    
    for i in range(1, n):
        dp[i] = max(arr[i], dp[i-1] + arr[i])
    
    return max(dp)
```

## 🚀 고급 DP 문제들

### 1. 0/1 배낭 문제
```python
def knapsack_01(weights, values, capacity):
    n = len(weights)
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]
    
    for i in range(1, n + 1):
        for w in range(capacity + 1):
            if weights[i-1] <= w:
                dp[i][w] = max(dp[i-1][w], 
                              dp[i-1][w - weights[i-1]] + values[i-1])
            else:
                dp[i][w] = dp[i-1][w]
    
    return dp[n][capacity]

# 공간 최적화 (1차원 배열)
def knapsack_01_optimized(weights, values, capacity):
    n = len(weights)
    dp = [0] * (capacity + 1)
    
    for i in range(n):
        for w in range(capacity, weights[i] - 1, -1):
            dp[w] = max(dp[w], dp[w - weights[i]] + values[i])
    
    return dp[capacity]
```

### 2. 최장 증가 부분 수열 (LIS)
```python
def longest_increasing_subsequence(arr):
    if not arr:
        return 0
    
    n = len(arr)
    dp = [1] * n
    
    for i in range(1, n):
        for j in range(i):
            if arr[i] > arr[j]:
                dp[i] = max(dp[i], dp[j] + 1)
    
    return max(dp)

# 이진 탐색을 이용한 최적화 (O(n log n))
def lis_optimized(arr):
    if not arr:
        return 0
    
    tails = [arr[0]]
    
    for num in arr[1:]:
        if num > tails[-1]:
            tails.append(num)
        else:
            # 이진 탐색으로 적절한 위치 찾기
            left, right = 0, len(tails) - 1
            while left < right:
                mid = (left + right) // 2
                if tails[mid] < num:
                    left = mid + 1
                else:
                    right = mid
            tails[left] = num
    
    return len(tails)
```

### 3. 편집 거리 (Edit Distance)
```python
def edit_distance(str1, str2):
    m, n = len(str1), len(str2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    # 초기화
    for i in range(m + 1):
        dp[i][0] = i
    for j in range(n + 1):
        dp[0][j] = j
    
    # DP 계산
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if str1[i-1] == str2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(dp[i-1][j],    # 삭제
                                  dp[i][j-1],      # 삽입
                                  dp[i-1][j-1])   # 교체
    
    return dp[m][n]
```

### 4. 동전 거스름돈
```python
def coin_change(coins, amount):
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] = min(dp[i], dp[i - coin] + 1)
    
    return dp[amount] if dp[amount] != float('inf') else -1

# 모든 방법의 수 계산
def coin_change_ways(coins, amount):
    dp = [0] * (amount + 1)
    dp[0] = 1
    
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] += dp[i - coin]
    
    return dp[amount]
```

## 🔧 2차원 DP 문제들

### 1. 최장 공통 부분 수열 (LCS)
```python
def longest_common_subsequence(text1, text2):
    m, n = len(text1), len(text2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i-1] == text2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    
    return dp[m][n]
```

### 2. 행렬 체인 곱셈
```python
def matrix_chain_multiplication(dims):
    n = len(dims) - 1
    dp = [[0] * n for _ in range(n)]
    
    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length - 1
            dp[i][j] = float('inf')
            
            for k in range(i, j):
                cost = dp[i][k] + dp[k+1][j] + dims[i] * dims[k+1] * dims[j+1]
                dp[i][j] = min(dp[i][j], cost)
    
    return dp[0][n-1]
```

### 3. 격자 경로 문제
```python
def unique_paths(m, n):
    dp = [[1] * n for _ in range(m)]
    
    for i in range(1, m):
        for j in range(1, n):
            dp[i][j] = dp[i-1][j] + dp[i][j-1]
    
    return dp[m-1][n-1]

# 공간 최적화
def unique_paths_optimized(m, n):
    dp = [1] * n
    
    for i in range(1, m):
        for j in range(1, n):
            dp[j] += dp[j-1]
    
    return dp[n-1]
```

### 2. 메모이제이션 패턴
```python
def solve_with_memoization(n, memo=None):
    if memo is None:
        memo = {}
    
    # 기저 사례
    if n <= 1:
        return n
    
    # 메모이제이션 체크
    if n in memo:
        return memo[n]
    
    # 재귀 호출
    result = solve_with_memoization(n-1, memo) + solve_with_memoization(n-2, memo)
    
    # 결과 저장
    memo[n] = result
    return result
```

def solve_with_tabulation(n):
    # DP 배열 초기화

    dp[0] = 0

### 1. 공간 최적화
```python
# 2차원 배열을 1차원으로 최적화
def optimized_2d_dp(n):
    dp = [0] * (n + 1)
    dp[0] = 1
    
    for i in range(1, n + 1):
        for j in range(i, n + 1):
            dp[j] += dp[j - i]
    
    return dp[n]
```

### 2. 슬라이딩 윈도우
```python
def sliding_window_dp(arr, k):
    n = len(arr)
    if n < k:
        return 0
    
    # 초기 윈도우
    window_sum = sum(arr[:k])
    max_sum = window_sum
    
    # 슬라이딩
    for i in range(k, n):
        window_sum = window_sum - arr[i-k] + arr[i]
        max_sum = max(max_sum, window_sum)
    
    return max_sum
```

### 3. 상태 압축
```python
def state_compression_dp(n):
    # 비트마스킹을 이용한 상태 압축
    dp = [0] * (1 << n)
    dp[0] = 1
    
    for state in range(1 << n):
        for i in range(n):
            if not (state & (1 << i)):
                dp[state | (1 << i)] += dp[state]
    
    return dp[(1 << n) - 1]
```
