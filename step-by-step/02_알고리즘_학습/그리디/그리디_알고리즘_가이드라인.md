# 그리디 알고리즘 학습 가이드라인

## 🎯 그리디 알고리즘의 중요성

그리디 알고리즘은 각 단계에서 가장 좋아 보이는 선택을 하는 방식으로, 최적화 문제를 해결하는 데 사용됩니다. 간단하면서도 강력한 알고리즘으로, 많은 코딩테스트 문제에서 활용됩니다.

## 📊 그리디 알고리즘의 특징

### 1. 장점
- **구현이 간단**: 직관적인 접근 방식
- **실행 속도가 빠름**: 각 단계에서 최적 선택
- **메모리 효율적**: 추가 공간이 거의 불필요

### 2. 단점
- **최적해 보장 안됨**: 항상 최적해를 찾지 못할 수 있음
- **증명이 어려움**: 그리디 선택의 정당성 증명 필요

### 3. 적용 가능한 경우
- **최적 부분 구조**: 전체 문제의 최적해가 부분 문제의 최적해로 구성
- **그리디 선택 성질**: 각 단계에서 최적 선택이 전체 최적해로 이어짐

### 1. 동전 거스름돈
```python
def coin_change_greedy(amount, coins):
    # 동전을 내림차순으로 정렬
    coins.sort(reverse=True)
    count = 0
    
    for coin in coins:
        if amount >= coin:
            count += amount // coin
            amount %= coin
    
    return count if amount == 0 else -1

# 최소 동전 개수로 거슬러주기
def min_coins_greedy(amount, coins):
    coins.sort(reverse=True)
    result = []
    
    for coin in coins:
        while amount >= coin:
            result.append(coin)
            amount -= coin
    
    return result if amount == 0 else []
```

### 2. 활동 선택 문제 (Activity Selection)

```python
def activity_selection(activities):
    # 종료 시간으로 정렬
    activities.sort(key=lambda x: x[1])
    
    selected = []
    last_end = 0
    
    for start, end in activities:
        if start >= last_end:
            selected.append((start, end))
            last_end = end
    
    return selected

# 회의실 배정 문제
def meeting_room_assignment(meetings):
    meetings.sort(key=lambda x: (x[1], x[0]))  # 종료 시간, 시작 시간 순
    
    count = 0
    last_end = 0
    
    for start, end in meetings:
        if start >= last_end:
            count += 1
            last_end = end
    
    return count
```

### 3. 분할 가능 배낭 문제
```python
def fractional_knapsack(weights, values, capacity):
    # 가치/무게 비율로 정렬
    items = [(values[i], weights[i], values[i]/weights[i]) 
             for i in range(len(weights))]
    items.sort(key=lambda x: x[2], reverse=True)
    
    total_value = 0
    remaining_capacity = capacity
    
    for value, weight, ratio in items:
        if remaining_capacity >= weight:
            total_value += value
            remaining_capacity -= weight
        else:
            total_value += ratio * remaining_capacity
            break
    
    return total_value
```

## 🚀 고급 그리디 문제들

### 1. 허프만 코딩 (Huffman Coding)
```python
import heapq

class HuffmanNode:
    def __init__(self, char, freq):
        self.char = char
        self.freq = freq
        self.left = None
        self.right = None
    
    def __lt__(self, other):
        return self.freq < other.freq

def build_huffman_tree(freq_dict):
    # 최소 힙 생성
    heap = [HuffmanNode(char, freq) for char, freq in freq_dict.items()]
    heapq.heapify(heap)
    
    # 트리 구성
    while len(heap) > 1:
        left = heapq.heappop(heap)
        right = heapq.heappop(heap)
        
        internal = HuffmanNode(None, left.freq + right.freq)
        internal.left = left
        internal.right = right
        
        heapq.heappush(heap, internal)
    
    return heap[0]

def get_huffman_codes(root, code="", codes=None):
    if codes is None:
        codes = {}
    
    if root.char is not None:
        codes[root.char] = code
    else:
        get_huffman_codes(root.left, code + "0", codes)
        get_huffman_codes(root.right, code + "1", codes)
    
    return codes
```

### 2. 작업 스케줄링
```python
def job_scheduling(jobs):
    # 마감일로 정렬
    jobs.sort(key=lambda x: x[1])
    
    schedule = []
    current_time = 0
    
    for job_id, deadline, profit in jobs:
        if current_time < deadline:
            schedule.append(job_id)
            current_time += 1
    
    return schedule

# 최대 이익을 위한 작업 스케줄링
def max_profit_job_scheduling(jobs):
    # 이익으로 정렬
    jobs.sort(key=lambda x: x[2], reverse=True)
    
    schedule = [None] * max(job[1] for job in jobs)
    total_profit = 0
    
    for job_id, deadline, profit in jobs:
        # 마감일부터 역순으로 빈 슬롯 찾기
        for i in range(deadline - 1, -1, -1):
            if schedule[i] is None:
                schedule[i] = job_id
                total_profit += profit
                break
    
    return [job for job in schedule if job is not None], total_profit
```

### 3. 최소 신장 트리 (Kruskal's Algorithm)
```python
class DisjointSet:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
    
    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]
    
    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py:
            return False
        
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        
        self.parent[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1
        return True

def kruskal_mst(edges, n):
    # 가중치로 정렬
    edges.sort(key=lambda x: x[2])
    
    ds = DisjointSet(n)
    mst = []
    total_weight = 0
    
    for u, v, weight in edges:
        if ds.union(u, v):
            mst.append((u, v, weight))
            total_weight += weight
    
    return mst, total_weight
```

### 4. 최단 경로 (Dijkstra's Algorithm)
```python
import heapq

def dijkstra(graph, start):
    n = len(graph)
    distances = [float('inf')] * n
    distances[start] = 0
    
    pq = [(0, start)]
    
    while pq:
        dist, node = heapq.heappop(pq)
        
        if dist > distances[node]:
            continue
        
        for neighbor, weight in graph[node]:
            new_dist = dist + weight
            if new_dist < distances[neighbor]:
                distances[neighbor] = new_dist
                heapq.heappush(pq, (new_dist, neighbor))
    
    return distances
```

### 시간복잡도
- **정렬**: O(n log n) - 대부분의 그리디 문제

- **전체**: O(n log n) - 정렬이 지배적

### 공간복잡도
- **O(1)**: 추가 공간 불필요
- **O(n)**: 결과 저장 필요
